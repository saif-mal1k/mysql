

------------------------------create new user---------------------------------------------------------------------------
mysql>CREATE USER 'newuser'@'myhost.example.com' IDENTIFIED BY 'yourpassword';
mysql>GRANT ALL ON databasename.* TO 'newuser'@'myhost.example.com';
mysql>REVOKE * ON databasename.* FROM 'newuser'@'myhost.example.com';

________________________________________________________________________________________________________________________

-----------------------------alter command on table---------------------------------------------------------------------
mysql>ALTER TABLE tablename DROP columnname;			//deletes column from table "temporarily"
mysql>ex:ALTER TABLE student MODIFY name varchar(20) not null;  //modify schema of table
mysql>ALTER TABLE tablename ADD columnname type(size); 		//add new column to table
________________________________________________________________________________________________________________________

------------------------------drop command on table---------------------------------------------------------------------
mysql>DROP TABLE tablename;					//delete table temporarily
________________________________________________________________________________________________________________________

------------------------------truncate command on table-----------------------------------------------------------------
mysql>TRUNCATE TABLE tablename;					//permanently delete the table
