

------------------------------create new user---------------------------------------------------------------------------
mysql>CREATE USER 'newuser'@'myhost.example.com' IDENTIFIED BY 'yourpassword';
mysql>GRANT ALL ON databasename.* TO 'newuser'@'myhost.example.com';
mysql>REVOKE * ON databasename.* FROM 'newuser'@'myhost.example.com';

-----------------------------create new database------------------------------------------------------------------------
mysql>CREATE DATABASE databasename;

________________________________________________________________________________________________________________________


-----------------------------create new table in a database-------------------------------------------------------------
mysql>SHOW databases;
mysql>USE databasename;
mysql>SHOW tables;
mysql>CREATE TABLE tablename(attrib VARCHAR(10),attrib2 INT(10));
mysql>INSERT INTO tablename(attrib,attrib2) VALUES('name',324342);
________________________________________________________________________________________________________________________


-----------------------------selecting elements from a table------------------------------------------------------------
mysql>USE databasename;
mysql>SELECT * FROM tablename;
mysql>SELECT * attribname FROM tablename;

________________________________________________________________________________________________________________________


------------------------------deleting elements from a table------------------------------------------------------------
mysql>USE databasename;
